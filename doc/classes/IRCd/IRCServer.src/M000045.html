  def handle_reply(sock, rpl)
    return unless rpl
    case rpl[0]
    when Array
      rpl.each{|rp|
        handle_reply(sock, rp)
      }
    when Command
      send_client_message(sock, @users[sock], *rpl)
    else
      send_server_message(sock, *rpl)
    end
  end
