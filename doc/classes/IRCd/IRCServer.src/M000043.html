  def print_motd(sock)
    send_server_message(sock, "375", "- #{config[:ServerName]} Message of the day - ")
    config[:Motd].each_line{|line|
      line.chomp!
      send_server_message(sock, "372", "- " + line)
    }
    send_server_message(sock, "376", ":End of MOTD command")
  end
