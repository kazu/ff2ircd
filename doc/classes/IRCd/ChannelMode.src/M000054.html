  def get_flags
    flg = ""
    flg << "n" if @n
    flg << "i" if @invite
    flg << "a" if @anony
    flg << "m" if @moderate
    flg << "q" if @quiet
    flg << "p" if @private
    flg << "s" if @secret
    flg << "k" unless @key.empty?
    flg << "t" if @topic_op_only
    flg << "l" if @user_max > -1
  end
