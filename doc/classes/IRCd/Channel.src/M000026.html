  def set_mode(who, flags, param)
    if @mode.op?(who)
      rpl = @mode.set_flags(who, flags, param)
      send_to_members(who, Command::MODE, @name, *rpl)
      nil
    elsif !joined?(who)
      return ["482", @name, "You're not channel operator"]
    else
      return ["442", @name, "You're not on that channel"]
    end
  end
