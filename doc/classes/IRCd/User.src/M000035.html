  def set_flags(*flags)
    rls = []
    flags.each{|flg|
      fl = flg[0,1]
      toggle = flg[0] == ?+
      flg[1..-1].each{|md|
        case md
        when "i"
          @invisible = toggle
          fl << "i"
        when "w"
          @wallop = toggle
          fl << "w"
        when"r"
          if toggle
            @restriction = true
            fl << "r"
          end
        when "o"
          unless toggle
            @operator = false
            fl << "o"
          end
        when "O"
          unless toggle
            @local_operator = false
            fl << "O"
          end
        when "s"
          @s = toggle
          fl << "s"
        end
      }
      rls << fl
    }
    rls
  end
