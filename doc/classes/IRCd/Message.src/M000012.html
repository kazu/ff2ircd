    def to_s
      str = ''
      if @prefix
        str << ':'
        str << @prefix
        str << ' '
      end

      str << @command

      if @params
        f = false
        @params.each do |param|
          param = param.to_s
          str << ' '
          if !f && (param.empty? || /[: ]/ =~ param)
            str << ':'
            str << param
            f = true
          else
            str << param
          end
        end
      end

      str << "\x0D\x0A"

      str
    end
