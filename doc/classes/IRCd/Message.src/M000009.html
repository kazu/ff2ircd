    def self.parse(msg)
      prefix = nil
      command = ""
      tmp = []
      if msg[0] == ?:
        prefix, command, *tmp = msg[1..-1].split(" ")
      else
        command, *tmp = msg.split(" ")
      end
      command
      params = []
      eoc = false
      tmp.each_with_index{|ar,ind|
        if eoc
          params[-1] += ar
        else
          if ar[0] == ?:
            eoc = true
            ar = ar[1..-1]
          end
          params << ar
        end
      }
      self.build(prefix, command, params)
    end
